NotiEntity

@Entity(tableName = "notis")
data class NotiEntity(
    @PrimaryKey val notiCode: String,
    val userId: String = "",
    val notiType: String = "",           // 카테고리 (mention/invite/system 등)
    val title: String = "",
    val contents: String = "",
    val linkUrl: String = "",            // 클릭 시 이동 URL (앱 내 딥링크)
    val sendDate: Long = 0L,
    val readDate: Long = 0L,             // 0=미읽음
    val state: Int = 0,                  // 0=정상, -1=삭제
    val additional: String = "{}"
)
